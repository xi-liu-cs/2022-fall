  \text{continuity/interchange of limits}\\
  \{x_n\}, x_n \rightarrow x \text{ as } n \rightarrow \infty\\
  \lim_{n \rightarrow \infty} g(x_n) = g(\lim_{n \rightarrow \infty} x_n)\\
  \int_a ^ b: \mathscr{R}[a, b] \rightarrow \mathbb{R}\\
  \lim_{n \rightarrow \infty} \int_a ^ b f_n = \int_a ^ b (\lim_{n \rightarrow \infty} f_n)\\
  \text{continuous functions on \mathscr{R}[a, b]}\\
  \text{metric spaces, generalize } |x_n - x + x - x_k| \leq |x_n - x| + |x - x_n|
